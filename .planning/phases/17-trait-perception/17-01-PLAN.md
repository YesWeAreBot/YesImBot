   ```ts
     dimension: string;
     value: string;
     confidence: number;
     metadata?: Record<string, unknown>;
   }
   ```

2. Create `core/src/services/trait/types.ts` with:
   - `TraitDetector` interface: `name: string`, `start(ctx: Context, analyzer: TraitAnalyzer): void`, `detect(scope: Scope, view: HorizonView): TraitSignal[]`
   - Import types from shared/types and horizon/types as needed
   - Use forward reference or import type for TraitAnalyzer to avoid circular

3. Create `core/src/services/trait/service.ts`:
   - Koishi Service subclass pattern (see CLAUDE.md): `class TraitAnalyzer extends Service`
   - Declaration merging: `Context { 'yesimbot.trait': TraitAnalyzer }`
   - Constructor: `super(ctx, 'yesimbot.trait', false)`, create logger once as `this.logger = ctx.logger('trait')`
   - `static inject = ['yesimbot.horizon']` — only depends on horizon for events
   - Private `detectors: TraitDetector[]` array
   - Private `stateStore: Map<string, unknown>` for per-channel state
   - `registerDetector(detector: TraitDetector)`: pushes to array, calls `detector.start(this.ctx, this)` — use a sub-context via `this.ctx.plugin()` or pass `this.ctx` so event listeners auto-dispose
   - `getState<T>(detectorName: string, channelKey: string): T | undefined` — reads from `stateStore` keyed by `${detectorName}:${channelKey}`
   - `setState<T>(detectorName: string, channelKey: string, state: T): void` — writes to stateStore
   - `async analyze(scope: Scope, view: HorizonView): Promise<TraitSignal[]>` — runs all detectors via `Promise.allSettled(this.detectors.map(d => Promise.resolve(d.detect(scope, view))))`, flattens fulfilled results, logs rejected ones
   - `protected async start()`: empty for now (detectors registered by Plan 02)
   - Export config type and schema (minimal — no config fields needed yet, but follow pattern)

4. Create `core/src/services/trait/index.ts`: re-export TraitAnalyzer, config type/schema, and types

IMPORTANT: Follow CLAUDE.md rules:
- Service subclass pattern, NOT ctx.provide()
- Logger created once in constructor, NOT chained on ctx.logger()
- No explicit `any` — use proper generics
- No duplicate config field (inherited from Service base)
