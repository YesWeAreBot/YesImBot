```typescript
// Escape dynamic values for XML attributes
const esc = (v: string) =>
  v.replace(/&/g, "&amp;").replace(/"/g, "&quot;").replace(/</g, "&lt;").replace(/>/g, "&gt;");

// For message type:
const hhmm = obs.timestamp.toTimeString().slice(0, 5);
// ... existing shortId, senderName, senderId logic ...
let attrs = `id="${shortId}" sender="${esc(senderName)}" senderId="${esc(senderId)}" time="${hhmm}"`;
if (obs.replyTo) {
  const replyShortId = this.getShortId(channelKey, obs.replyTo);
  if (replyShortId !== undefined) attrs += ` replyTo="${replyShortId}"`;
}
return `<msg ${attrs}>${obs.content}</msg>`;
```

Apply `esc()` to `senderName` and `senderId` attribute values (Pitfall 5 from research — user names can contain `"` or `>`). Content is already pre-escaped by ElementFormatterService.

2. **AgentAction observations** — new rendering as `<bot-action>` XML tag:

```typescript
if (obs.type === "agent.action") {
  const d = obs.data;
  const triggerAttr = d.triggerMsgId
    ? ` trigger="#${this.getShortId(channelKey ?? "", d.triggerMsgId) ?? "?"}"`
    : "";
  const lines = d.actions.map((a) => {
    const r = d.toolResults.find((t) => t.name === a.name);
    if (a.name === "send_message") {
      const ok = r?.status === "ok" || r?.status === "fulfilled" || (r != null && !r.error);
      return ok ? "send_message -> sent" : `send_message -> failed: ${r?.error ?? "unknown"}`;
    }
    const status = r ? r.status + (r.error ? ": " + r.error : "") : "no result";
    const preview = r?.result != null ? String(r.result).slice(0, 200) : "";
    return `${a.name}(${JSON.stringify(a.params ?? {})}) -> ${status}${preview ? ": " + preview : ""}`;
  });
  return `<bot-action round="${d.round}"${triggerAttr}>${lines.join("; ")}</bot-action>`;
}
```

3. **AgentResponse observations** — only render if there's an error (network/parse failure). Normal successful responses are captured by AgentAction. For backward compat, old rows with `actions` array are handled by the AgentAction branch (Plan 01's toObservations emits both). New slim rows with only `rawText` and no error can be omitted from rendering (the LLM doesn't need to see its own raw JSON output):

```typescript
if (obs.type === "agent.response") {
  const d = obs.data;
  if (d.error) {
    return `<bot-error round="${d.round}">${d.error}</bot-error>`;
  }
  // Old rows with actions (backward compat) — render as bot-action
  if (d.actions?.length) {
    const hhmm = obs.timestamp.toTimeString().slice(0, 5);
    const sendAction = d.actions.find((a) => a.name === "send_message");
    const otherTools = d.actions.filter((a) => a.name !== "send_message").map((a) => a.name);
    if (sendAction) {
      const content = (sendAction.params?.content as string) ?? "";
      const suffix = otherTools.length ? ` [also: ${otherTools.join(", ")}]` : "";
      return `[${hhmm}] [Bot]: ${content}${suffix}`;
    }
    if (d.actions.length === 0) return `[${hhmm}] [Bot]: (chose silence)`;
    return `[${hhmm}] [Bot Action]: ${d.actions.map((a) => a.name).join(", ")}`;

  // Successful response with no error and no actions (new shape) — skip rendering

}
```
